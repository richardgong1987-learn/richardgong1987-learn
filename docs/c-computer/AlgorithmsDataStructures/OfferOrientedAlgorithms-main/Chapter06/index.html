public int largestRectangleArea(int[] heights) {    if (heights.length == 0) {        return 0;    }
    return helper(heights, 0, heights.length);}
private int helper(int[] heights, int start, int end) {    if (start == end) {        return 0;    }
    if (start + 1 == end) {        return heights[start];    }
    int minIndex = start;    for (int i = start + 1; i < end; i++) {        if (heights[i] < heights[minIndex]) {            minIndex = i;        }    }
    int area = (end - start) * heights[minIndex];    int left = helper(heights, start, minIndex);    int right = helper(heights, minIndex + 1, end);
    area = Math.max(area, left);    return Math.max(area, right);}

public int largestRectangleArea(int[] heights) {    Stack<Integer> stack = new Stack<>();    stack.push(-1);
    int maxArea = 0;    for (int i = 0; i < heights.length; i++) {        while (stack.peek() != -1            && heights[stack.peek()] >= heights[i]) {            int height = heights[stack.pop()];            int width = i - stack.peek() - 1;            maxArea = Math.max(maxArea, height * width);        }
        stack.push(i);    }
